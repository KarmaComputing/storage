How to mount

    apt-get install cifs-utils

    mkdir /mnt/mystorage

    # Create your keyring file
    cat << EOF >> /etc/ceph/ceph.client.$USERNAME.keyring
    {{ credentials }}
    EOF

    mount -t ceph $CEPH_MON_IP:/$USERNAME /mnt/mystorage -o name=$USERNAME,secret=$SECRET
